CREATE VIEW VW_RESULTS AS
	SELECT COUNTRY, YEAR, VALUE_1, VALUE_2, LAST_UPDATE
	FROM (
      	SELECT COUNTRY, YEAR, MAX(CREATED_DATE) as LAST_UPDATE
      	FROM RESULTS
      	GROUP BY COUNTRY, YEAR
	) LASTDATE
	INNER JOIN RESULTS LASTDATE
	ON RESULTS.COUNTRY = LASTDATE.COUNTRY AND RESULTS.CREATED_DATE = LASTDATE.LAST_UPDATE AND RESULTS.YEAR = LASTDATE.YEAR
